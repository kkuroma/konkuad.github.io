# ฟังก์ชันเรียกชุดข้อมูลเพื่อใช้ในขั้นตอนถัดไป

def getdata(N=100,mean=0,stdev=1,MIN=-1,MAX=1):
    dataset = np.random.standard_normal(N) #random
    dataset = dataset*stdev #adjust to stdev
    dataset = dataset+mean #adjust to mean
    dataset = np.array([max(MIN,i) for i in dataset]) #set minimal value
    dataset = np.array([min(MAX,i) for i in dataset]) #set maximal value
    return dataset

ฟังก์ชันข้างต้นจะสุ่มค่าที่กำหนดค่าเฉลี่ย ส่วนเบี่ยงเบนมาตรฐาน และค่าต่ำสุดสูงสุด ยกตัวอย่างเช่น

Skewed Gaussian Distribution¶

คือ การกระจายตัวข้อมูลที่เบ้ไปทางใดทางหนึ่ง นั่นคือไม่ใช่ gaussian distribution ปกติ

In [59]:
def getskewdata(N=100,mean=0,stdev=1,MIN=-1,MAX=1,alpha=1):
    dataset = stats.skewnorm.rvs(alpha, size=N) #random
    dataset = dataset*stdev #adjust to stdev
    dataset = dataset+mean #adjust to mean
    dataset = np.array([max(MIN,i) for i in dataset]) #set minimal value
    dataset = np.array([min(MAX,i) for i in dataset]) #set maximal value
    return dataset

เบ้ซาย คือ ข้อมูลไปกองกันทางขวา (โดนปัดจากทางซ้าย)

ความเบี่ยงเบนสัมบูรณ์ (median absolute deviation - MAD)¶

คือ ค่าเฉลี่ยระยะห่างระหว่างข้อมูลทุกค่ากับค่ามัธยฐาน ในบางทีดีกว่า sd ถ้าข้อมูลเบ้มากๆ

In [136]:
def MAD(dataset):
    median = np.median(dataset)
    deviation = dataset-median
    return np.mean(np.abs(deviation))

def CI(accuracy):
    ci = stats.norm.interval(0.95, loc=np.mean(accuracy), scale=np.std(accuracy))
    return 0.5*(ci[1]-ci[0])

  • z คือ ค่าความมั่นใจที่อยากได้
  • s คือ stdev
  • n คือ sample size

Case 4 - One Way Analysis of Variance (ANOVA)¶

ศึกษาความแตกต่างของค่าจากหลายๆ กลุ่มตัวอย่าง หรือค่ามากกว่าหรือเท่ากับ 2 ค่าจากกลุ่มตัวอย่างเดียวกัน ว่าแตกต่างกันนัยสำคัญหรือไม่ สูตรค่อนข้างซับซ้อนจึงไม่ได้ใส่ไว้

เราจะศึกษาว่า คะแนนวิชาเลข ฟิสิกส์ เคมี ชีวะ มีวิชาไหนที่ต่างจากวิชาอื่นอย่างนัยสำคัญไหม

In [216]:
chem_score = getdata(N = 100, mean = 90, stdev = 3.5, MIN = 0, MAX =100)
bio_score = getdata(N = 100, mean = 91, stdev = 3.2, MIN = 0, MAX =100)
math_score = getdata(N = 100, mean = 92, stdev = 3.4, MIN = 0, MAX =100)
phys_score = getdata(N = 100, mean = 91.5, stdev = 3.0, MIN = 0, MAX =100)

anova = stats.f_oneway(chem_score, bio_score, math_score, phys_score)
print(f'pvalue = {round(pvalue,2)}, which means the scores are different at a confidence value of {round((1-pvalue)*100,2)} percent')
